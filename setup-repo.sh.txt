#!/bin/bash

# ============================================
# LexIA Platform - Setup Automático
# Execute: bash setup-repo.sh
# ============================================

set -e  # Parar se houver erro

echo "🚀 Iniciando setup do LexIA Platform..."

# Cores
GREEN='\033[0;32m'
BLUE='\033[0;34m'
RED='\033[0;31m'
NC='\033[0m'

# Perguntar informações
echo -e "\n${BLUE}=== CONFIGURAÇÃO INICIAL ===${NC}"
read -p "Qual é seu usuário do GitHub? (ex: kelsonmagalhaes): " GITHUB_USER
read -p "Qual é seu email do GitHub? (ex: seu-email@example.com): " GITHUB_EMAIL
read -p "Qual é seu token do GitHub? (gere em: https://github.com/settings/tokens): " GITHUB_TOKEN

# Configurar Git globalmente
git config --global user.name "$GITHUB_USER"
git config --global user.email "$GITHUB_EMAIL"

echo -e "${GREEN}✅ Git configurado!${NC}"

# Criar diretório
PROJECT_DIR="lexia-platform"
if [ -d "$PROJECT_DIR" ]; then
    echo -e "${RED}❌ Pasta $PROJECT_DIR já existe!${NC}"
    read -p "Deletar e recrear? (s/n): " DELETE
    if [ "$DELETE" = "s" ]; then
        rm -rf "$PROJECT_DIR"
    else
        exit 1
    fi
fi

echo -e "\n${BLUE}=== CRIANDO ESTRUTURA ===${NC}"

# Criar pasta
mkdir -p "$PROJECT_DIR"
cd "$PROJECT_DIR"

# Inicializar Git
git init
git config user.name "$GITHUB_USER"
git config user.email "$GITHUB_EMAIL"

echo -e "${GREEN}✅ Git inicializado${NC}"

# ============================================
# CRIAR ARQUIVOS
# ============================================

echo -e "\n${BLUE}=== CRIANDO ARQUIVOS ===${NC}"

# Criar pastas
mkdir -p src/{pages,components/{layout,home,quiz,performance,ui},lib,api,hooks,utils,test/__tests__}
mkdir -p public
mkdir -p .github/workflows

echo -e "${GREEN}✅ Pastas criadas${NC}"

# ============================================
# package.json
# ============================================
cat > package.json << 'EOF'
